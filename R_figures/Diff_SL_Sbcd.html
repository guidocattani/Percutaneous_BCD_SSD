read_diff_SL <- function(path_in) {
  dt <- read.csv(path_in)
  dt <- dt[ -c(2, 3, 4, 5, 6, 7, 19, 20)]
  # rename columns
  clm = colnames(dt)
  clm= clm[-c(1)]
  newco <- c()
  for (fb in clm){
      f <- strsplit(fb, "_")[[1]][2]
      newco <- c(newco, f)
  }   

newco <- c('Study_ID', newco)
print(newco)
colnames(dt) <- newco
  return(dt)
}
