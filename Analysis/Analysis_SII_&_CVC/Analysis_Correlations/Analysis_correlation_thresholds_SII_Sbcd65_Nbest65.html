# lin. reg. SII combination path (si) function thresholds (df)
def lreg_SII(thr, si):
    l = list(thr.columns)
    d = dict()
    for j in range(6):
        thres = (thr.iloc[:, j]).to_numpy(dtype='float')
        lrgrs = linregress(thres, si)
        r = lrgrs[2]
        r2 = round(r**2, 3)
        r = round(r, 3)
        p = round(lrgrs[3], 3)
        se = round(lrgrs[4], 4)
        print(l[j], 'r square =', r2, ' r =', r, ' p =', p, ' se =', se)
        de = {l[j] : (r, r2)}
        d.update(de)
    df = pd.DataFrame.from_dict(d, orient='index')
    return df
